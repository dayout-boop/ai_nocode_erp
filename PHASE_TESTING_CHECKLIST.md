### 목표

### Step 1: 코드 수정 (30분)

#### server/_core/customOAuth.ts 수정
```typescript
export function getManusLoginUrlWithInvitation(
  invitationCode: string,
  skipPopup: boolean = true
): string {
  const params = new URLSearchParams({
    code: invitationCode,
    ...(skipPopup && { skip_credit_popup: 'true' }),
  });
  return `https://manus.im/login?${params.toString()}`;
}
```

#### client/src/pages/Partner/PartnerCustomLogin.tsx 수정
```typescript
const handleOAuthLogin = async (provider: 'google' | 'microsoft' | 'facebook' | 'apple') => {
  const response = await fetch(`/api/oauth/custom-login/${provider}?skipPopup=true`);
  const { url } = await response.json();
  window.open(url, '_blank', 'width=500,height=600');
};
```
